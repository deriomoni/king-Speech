let hasGyro = false;

function initGyro() {
  if (typeof DeviceOrientationEvent !== 'undefined' &&
      typeof DeviceOrientationEvent.requestPermission === 'function') {
    document.addEventListener('touchstart', function iosGyroRequest() {
      DeviceOrientationEvent.requestPermission().then(state => {
        if (state === 'granted') { bindGyro(); }
      }).catch(()=>{});
      document.removeEventListener('touchstart', iosGyroRequest);
    }, { once: true });
  } else {
    bindGyro();
  }
}

function bindGyro() {
  window.addEventListener('deviceorientation', (e) => {
    if (e.gamma === null && e.beta === null) return;
    hasGyro = true;
    const gx = (e.gamma || 0) / 45;
    const gy = ((e.beta || 0) - 45) / 45;
    targetParallaxX = Math.max(-1, Math.min(1, gx));
    targetParallaxY = Math.max(-1, Math.min(1, gy));
  }, true);
}

function initMouseParallax() {
  window.addEventListener('mousemove', (e) => {
    if (hasGyro) return;
    targetParallaxX = (e.clientX / window.innerWidth - 0.5) * 2;
    targetParallaxY = (e.clientY / window.innerHeight - 0.5) * 2;
  });
}

// ========== HELPERS ==========
function lerp(a, b, t) { return a + (b - a) * t; }
function easeInOutCubic(t) { return t < 0.5 ? 4*t*t*t : 1 - Math.pow(-2*t+2,3)/2; }

function getColorForKey(key) {
  const from = COLORS[currentIndex];
  const to = COLORS[targetIndex];
  const t = easeInOutCubic(morphProgress);
  const tb = themeBlend;

  function blend(dkF, ltF, dkT, ltT) {
    const dR = lerp(dkF[0], dkT[0], t), dG = lerp(dkF[1], dkT[1], t), dB = lerp(dkF[2], dkT[2], t);
    const lR = lerp(ltF[0], ltT[0], t), lG = lerp(ltF[1], ltT[1], t), lB = lerp(ltF[2], ltT[2], t);
    return { r: lerp(lR, dR, tb), g: lerp(lG, dG, tb), b: lerp(lB, dB, tb) };
  }

  switch(key) {
    case 'main': return blend(from.dark, from.light, to.dark, to.light);
    case 'a1': return blend(from.a1d, from.a1l, to.a1d, to.a1l);
    case 'a2': return blend(from.a2d, from.a2l, to.a2d, to.a2l);
    case 'a3': return blend(from.a3d, from.a3l, to.a3d, to.a3l);
    case 'a4': return blend(from.a4d, from.a4l, to.a4d, to.a4l);
    case 'comp': return blend(from.compD, from.compL, to.compD, to.compL);
    case 'core': {
      const m = blend(from.dark, from.light, to.dark, to.light);
      const lift = lerp(0.18, 0.12, tb);
      return {
        r: Math.min(255, m.r + (255 - m.r) * lift),
        g: Math.min(255, m.g + (255 - m.g) * lift),
        b: Math.min(255, m.b + (255 - m.b) * lift),
      };
    }
    default: return blend(from.dark, from.light, to.dark, to.light);
  }
}

// ========== OFFSCREEN CANVASES ==========
let offCanvas, offCtx, bloomCanvas, bloomCtx;
function initOffscreens() {
  offCanvas = document.createElement('canvas');
  offCanvas.width = W;
  offCanvas.height = H;
  offCtx = offCanvas.getContext('2d');

  // Bloom at quarter resolution for performance + natural blur
  bloomCanvas = document.createElement('canvas');
  bloomCanvas.width = Math.round(W / 4);
  bloomCanvas.height = Math.round(H / 4);
  bloomCtx = bloomCanvas.getContext('2d');
}

// ========== DRAW A SINGLE BLOB ==========
function drawBlob(context, px, py, radius, r, g, b, alpha) {
  const grad = context.createRadialGradient(px, py, 0, px, py, radius);
  grad.addColorStop(0, `rgba(${r},${g},${b},${alpha})`);
  grad.addColorStop(0.08, `rgba(${r},${g},${b},${alpha * 0.92})`);
  grad.addColorStop(0.2, `rgba(${r},${g},${b},${alpha * 0.7})`);
  grad.addColorStop(0.4, `rgba(${r},${g},${b},${alpha * 0.35})`);
  grad.addColorStop(0.65, `rgba(${r},${g},${b},${alpha * 0.10})`);
  grad.addColorStop(0.85, `rgba(${r},${g},${b},${alpha * 0.02})`);
  grad.addColorStop(1, `rgba(${r},${g},${b},0)`);
  context.fillStyle = grad;
  context.fillRect(0, 0, context.canvas.width, context.canvas.height);
}

// ========== MAIN DRAW ==========
function drawFrame() {
  time += 1;

  // Morph
  if (morphProgress < 1) {
    morphProgress = Math.min(1, morphProgress + morphSpeed);
    if (morphProgress >= 1) { currentIndex = targetIndex; }
  }

  // Theme blend
  themeBlend += (themeTarget - themeBlend) * 0.05;

  // Parallax smooth
  parallaxX += (targetParallaxX - parallaxX) * parallaxSmooth;
  parallaxY += (targetParallaxY - parallaxY) * parallaxSmooth;

  // Breathing
  const b1 = Math.sin(time * 0.005) * 0.06;
  const b2 = Math.sin(time * 0.003 + 1.5) * 0.04;
  const b3 = Math.sin(time * 0.008 + 3.0) * 0.03;
  const breathScale = 1 + b1 + b2 + b3;
  const baseRadius = Math.max(W, H) * breathScale;

  // BG
  const bgR = Math.round(lerp(255, 8, themeBlend));
  const bgG = Math.round(lerp(252, 6, themeBlend));
  const bgB = Math.round(lerp(248, 12, themeBlend));

  // ===== RENDER BLOBS ON OFFSCREEN (black bg, additive) =====
  offCtx.fillStyle = '#000';
  offCtx.fillRect(0, 0, W, H);
  offCtx.save();
  offCtx.globalCompositeOperation = 'lighter';

  for (const src of LIGHT_SOURCES) {
    const [dx, dy] = src.drift(time);
    const px = (src.basePos[0] + dx + parallaxX * parallaxStrength * src.pxMul) * W;
    const py = (src.basePos[1] + dy + parallaxY * parallaxStrength * src.pxMul) * H;
    const col = getColorForKey(src.colorKey);
    const r = Math.round(Math.min(255, col.r));
    const g = Math.round(Math.min(255, col.g));
    const b = Math.round(Math.min(255, col.b));
    const radius = baseRadius * src.rMul;
    const alpha = lerp(src.lightOp, src.darkOp, themeBlend);

    drawBlob(offCtx, px, py, radius, r, g, b, alpha);
  }
  offCtx.restore();

  // ===== BLOOM PASS: downscale → draw → upscale (natural box blur) =====
  const bW = bloomCanvas.width, bH = bloomCanvas.height;
  bloomCtx.fillStyle = '#000';
  bloomCtx.fillRect(0, 0, bW, bH);
  // Draw offscreen scaled down (acts as blur)
  bloomCtx.drawImage(offCanvas, 0, 0, bW, bH);
  // Extra softening: draw it on itself offset slightly multiple times
  bloomCtx.save();
  bloomCtx.globalCompositeOperation = 'lighter';
  bloomCtx.globalAlpha = 0.35;
  const bloomSpread = Math.round(bW * 0.02);
  bloomCtx.drawImage(bloomCanvas, -bloomSpread, -bloomSpread, bW + bloomSpread*2, bH + bloomSpread*2);
  bloomCtx.drawImage(bloomCanvas, bloomSpread, bloomSpread, bW - bloomSpread*2, bH - bloomSpread*2);
  bloomCtx.restore();

  // ===== COMPOSITE TO MAIN CANVAS =====
  ctx.fillStyle = `rgb(${bgR},${bgG},${bgB})`;
  ctx.fillRect(0, 0, W, H);

  const isLight = themeBlend < 0.5;
  const lightFactor = 1 - themeBlend; // 0 when dark, 1 when light

  if (themeBlend > 0.95) {
    // === DARK THEME ===
    // Additive blending (lights on dark)
    ctx.save();
    ctx.globalCompositeOperation = 'lighter';
    ctx.drawImage(offCanvas, 0, 0);
    ctx.restore();

    // Bloom layer
    ctx.save();
    ctx.globalCompositeOperation = 'lighter';
    ctx.globalAlpha = 0.3;
    ctx.drawImage(bloomCanvas, 0, 0, W, H);
    ctx.restore();

  } else if (themeBlend < 0.05) {
    // === LIGHT THEME ===
    // Strategy: use 'multiply' blend. On white bg, multiply with the blob colors
    // shows the actual colors. But blobs are on black = mostly dark.
    // So we invert: draw blobs as-is with normal blending but high alpha directly.
    // Better approach: draw the colored blobs DIRECTLY on white bg using normal composite.

    // Draw blobs directly onto the white canvas (no offscreen tricks)
    ctx.save();
    ctx.globalCompositeOperation = 'source-over';
    for (const src of LIGHT_SOURCES) {
      const [dx, dy] = src.drift(time);
      const px = (src.basePos[0] + dx + parallaxX * parallaxStrength * src.pxMul) * W;
      const py = (src.basePos[1] + dy + parallaxY * parallaxStrength * src.pxMul) * H;
      const col = getColorForKey(src.colorKey);
      const r = Math.round(Math.min(255, col.r));
      const g = Math.round(Math.min(255, col.g));
      const b = Math.round(Math.min(255, col.b));
      const radius = baseRadius * src.rMul;
      const alpha = src.lightOp * 0.85;

      drawBlob(ctx, px, py, radius, r, g, b, alpha);
    }
    ctx.restore();

    // Bloom on light: draw bloom with multiply to tint white areas
    ctx.save();
    ctx.globalCompositeOperation = 'source-over';
    ctx.globalAlpha = 0.25;
    // Re-render bloom blobs directly too
    for (const src of LIGHT_SOURCES) {
      const [dx, dy] = src.drift(time);
      const px = (src.basePos[0] + dx + parallaxX * parallaxStrength * src.pxMul) * W;
      const py = (src.basePos[1] + dy + parallaxY * parallaxStrength * src.pxMul) * H;
      const col = getColorForKey(src.colorKey);
      const r = Math.round(Math.min(255, col.r));
      const g = Math.round(Math.min(255, col.g));
      const b = Math.round(Math.min(255, col.b));
      const radius = baseRadius * src.rMul * 1.4; // wider for bloom
      const alpha = src.bloomMul * 0.5;

      drawBlob(ctx, px, py, radius, r, g, b, alpha);
    }
    ctx.restore();

  } else {
    // === TRANSITION: blend between dark and light approaches ===

    // Dark contribution (additive)
    ctx.save();
    ctx.globalCompositeOperation = 'lighter';
    ctx.globalAlpha = themeBlend;
    ctx.drawImage(offCanvas, 0, 0);
    ctx.restore();

    // Light contribution (direct blobs)
    ctx.save();
    ctx.globalCompositeOperation = 'source-over';
    ctx.globalAlpha = lightFactor;
    for (const src of LIGHT_SOURCES) {
      const [dx, dy] = src.drift(time);
      const px = (src.basePos[0] + dx + parallaxX * parallaxStrength * src.pxMul) * W;
      const py = (src.basePos[1] + dy + parallaxY * parallaxStrength * src.pxMul) * H;
      const col = getColorForKey(src.colorKey);
      const r = Math.round(Math.min(255, col.r));
      const g = Math.round(Math.min(255, col.g));
      const b = Math.round(Math.min(255, col.b));
      const radius = baseRadius * src.rMul;
      const alpha = src.lightOp * 0.85;

      drawBlob(ctx, px, py, radius, r, g, b, alpha);
    }
    ctx.restore();

    // Bloom for both
    ctx.save();
    ctx.globalCompositeOperation = 'lighter';
    ctx.globalAlpha = 0.25 * themeBlend;
    ctx.drawImage(bloomCanvas, 0, 0, W, H);
    ctx.restore();
  }

  // Subtle top vignette
  const vigAlpha = lerp(0.03, 0.15, themeBlend);
  const vig = ctx.createLinearGradient(0, 0, 0, H * 0.35);
  vig.addColorStop(0, themeBlend > 0.5 ? `rgba(0,0,0,${vigAlpha})` : `rgba(200,200,210,${vigAlpha * 0.5})`);
  vig.addColorStop(1, 'rgba(0,0,0,0)');
  ctx.fillStyle = vig;
  ctx.fillRect(0, 0, W, H * 0.35);

  requestAnimationFrame(drawFrame);
}

// ========== UI ==========
function buildPicker() {
  const picker = document.getElementById('colorPicker');
  picker.innerHTML = '';
  COLORS.forEach((c, i) => {
    const dot = document.createElement('div');
    dot.className = 'color-dot' + (i === currentIndex ? ' active' : '');
    const col = c.dark;
    dot.style.background = `rgb(${col[0]},${col[1]},${col[2]})`;
    dot.style.boxShadow = `0 2px 8px rgba(${col[0]},${col[1]},${col[2]},0.45)`;
    dot.addEventListener('click', () => selectColor(i));
    picker.appendChild(dot);
  });
}

function selectColor(index) {
  if (index === targetIndex && morphProgress >= 1) return;
  currentIndex = targetIndex;
  targetIndex = index;
  morphProgress = 0;
  document.getElementById('colorName').textContent = COLORS[index].name;
  document.querySelectorAll('.color-dot').forEach((dot, i) => {
    dot.classList.toggle('active', i === index);
  });
}

function toggleTheme() {
  isDark = !isDark;
  themeTarget = isDark ? 1 : 0;
  document.body.classList.toggle('light-theme', !isDark);
  document.getElementById('themeIcon').textContent = isDark ? '🌙' : '☀️';
  document.getElementById('themeLabel').textContent = isDark ? 'Dark' : 'Light';
}
