```typescript

// На:

```

#### Добавить useEffect для countdown (сразу после объявления countdown):
```typescript
useEffect(() => {
  if (stage !== "countdown") return;
  if (countdown === 0) {
    // Запустить игру
    setStage("active");
    isPlayingRef.current = true;
    startRecognition();
    return;

  const id = setTimeout(() => setCountdown((c) => c - 1), 1000);
```
