<script>
/* ════════════════════════════════════════
   BACKGROUND — fluid blobs
════════════════════════════════════════ */
(function () {
  const canvas = document.getElementById('bgCanvas');
  const phone  = document.getElementById('phone');
  const ctx    = canvas.getContext('2d');

  function resize() {
    canvas.width  = phone.offsetWidth;
    canvas.height = phone.offsetHeight;
  }
  resize();
  window.addEventListener('resize', resize);

  const W = () => canvas.width;
  const H = () => canvas.height;

  /* palette */
  const COLORS = [
    { r: 38,  g: 6,   b: 80  },
    { r: 70,  g: 15,  b: 140 },
    { r: 100, g: 30,  b: 170 },
    { r: 130, g: 35,  b: 155 },
    { r: 60,  g: 18,  b: 110 },
  ];

  /* blobs */
  class Blob {
    constructor(seed) {
      this.reset(seed);
    }
    reset(seed = 0) {
      const w = W(), h = H();
      this.x  = w * (0.1 + Math.random() * 0.8);
      this.y  = h * (0.1 + Math.random() * 0.8);
      this.vx = (Math.random() - 0.5) * 0.35;
      this.vy = (Math.random() - 0.5) * 0.35;
      this.r  = w * (0.28 + Math.random() * 0.35);
      this.col = COLORS[Math.floor(Math.random() * COLORS.length)];
      this.alpha = 0.18 + Math.random() * 0.22;
      this.phase = Math.random() * Math.PI * 2;
      this.morphSpeed = 0.004 + Math.random() * 0.006;
      this.morphAmp   = 0.08 + Math.random() * 0.14;
    }
    update(t) {
      this.x += this.vx;
      this.y += this.vy;
      const w = W(), h = H();
      if (this.x < -this.r * 0.5 || this.x > w + this.r * 0.5) this.vx *= -1;
      if (this.y < -this.r * 0.5 || this.y > h + this.r * 0.5) this.vy *= -1;
      this.currentR = this.r * (1 + this.morphAmp * Math.sin(t * this.morphSpeed + this.phase));
    }
    draw(ctx) {
      const { r, g, b } = this.col;
      const grd = ctx.createRadialGradient(this.x, this.y, 0, this.x, this.y, this.currentR);
      grd.addColorStop(0,   `rgba(${r},${g},${b},${this.alpha})`);
      grd.addColorStop(0.5, `rgba(${r},${g},${b},${this.alpha * 0.5})`);
      grd.addColorStop(1,   `rgba(${r},${g},${b},0)`);
      ctx.beginPath();
      ctx.arc(this.x, this.y, this.currentR, 0, Math.PI * 2);
      ctx.fillStyle = grd;
      ctx.fill();
    }
  }

  const blobs = Array.from({ length: 8 }, (_, i) => new Blob(i));

  /* floating particles */
  const particles = Array.from({ length: 28 }, () => ({
    x: Math.random() * 400,
    y: Math.random() * 900,
    r: 0.6 + Math.random() * 1.2,
    a: 0.025 + Math.random() * 0.055,
    speed: 0.1 + Math.random() * 0.18,
    dx: (Math.random() - 0.5) * 0.15,
  }));

  let t = 0;

  function drawFrame() {
    t++;
    const w = W(), h = H();
    ctx.clearRect(0, 0, w, h);

    /* deep base gradient */
    const bg = ctx.createLinearGradient(0, 0, w * 0.6, h);
    bg.addColorStop(0,   '#0d0520');
    bg.addColorStop(0.4, '#130830');
    bg.addColorStop(1,   '#0a0318');
    ctx.fillStyle = bg;
    ctx.fillRect(0, 0, w, h);

    /* blobs */
    ctx.globalCompositeOperation = 'screen';
    blobs.forEach(b => { b.update(t); b.draw(ctx); });
    ctx.globalCompositeOperation = 'source-over';

    /* vignette — stronger for a more contained, minimal look */
    const vig = ctx.createRadialGradient(w / 2, h / 2, h * 0.1, w / 2, h / 2, h * 0.62);
    vig.addColorStop(0, 'rgba(0,0,0,0)');
    vig.addColorStop(1, 'rgba(0,0,0,0.88)');
    ctx.fillStyle = vig;
    ctx.fillRect(0, 0, w, h);

    /* floating particles */
    particles.forEach(p => {
      p.y -= p.speed;
      p.x += p.dx;
      if (p.y < -4)  p.y = h + 4;
      if (p.x < 0)   p.x = w;
      if (p.x > w)   p.x = 0;
      ctx.beginPath();
      ctx.arc(p.x, p.y, p.r, 0, Math.PI * 2);
      ctx.fillStyle = `rgba(200,170,255,${p.a})`;
      ctx.fill();
    });

    requestAnimationFrame(drawFrame);
  }
  drawFrame();
})();


/* ════════════════════════════════════════
   SPINNER — particle ring
════════════════════════════════════════ */
(function () {
  const canvas = document.getElementById('spinnerCanvas');
  const ctx    = canvas.getContext('2d');
  const CX = canvas.width  / 2;
  const CY = canvas.height / 2;

  const RADIUS      = 38;
  const PARTICLE_N  = 0;

  /* arc segments — simplified to two */
  const SEGMENTS = [
    { start: 0,    len: 0.55, width: 2.4, blur: 6,  colorA: '#ffffff', colorB: '#60cfff' },
    { start: 0.65, len: 0.18, width: 1.4, blur: 3,  colorA: 'rgba(160,120,255,0.7)', colorB: 'rgba(100,80,200,0)' },
  ];

  /* particle definitions around ring */
  const ringParticles = Array.from({ length: PARTICLE_N }, (_, i) => {
    const base = (i / PARTICLE_N) * Math.PI * 2;
    return {
      baseAngle: base,
      radius:    RADIUS + (Math.random() - 0.5) * 6,
      size:      0.7 + Math.random() * 1.8,
      speed:     0.008 + Math.random() * 0.006,
      phase:     Math.random() * Math.PI * 2,
      brightness: 0.3 + Math.random() * 0.7,
    };
  });

  const swirlDots = [];

  let angle = 0;
  let t     = 0;

  function drawFrame() {
    t++;
    angle += 0.022;

    ctx.clearRect(0, 0, canvas.width, canvas.height);

    /* ── outer glass rim ── */
    ctx.save();
    ctx.beginPath();
    ctx.arc(CX, CY, RADIUS + 7, 0, Math.PI * 2);
    const rimGrad = ctx.createLinearGradient(CX - RADIUS, CY - RADIUS, CX + RADIUS, CY + RADIUS);
    rimGrad.addColorStop(0,    'rgba(255,255,255,0.13)');
    rimGrad.addColorStop(0.45, 'rgba(255,255,255,0.04)');
    rimGrad.addColorStop(1,    'rgba(255,255,255,0.10)');
    ctx.strokeStyle = rimGrad;
    ctx.lineWidth   = 1;
    ctx.stroke();
    ctx.restore();

    /* ── inner glass rim ── */
    ctx.save();
    ctx.beginPath();
    ctx.arc(CX, CY, RADIUS - 7, 0, Math.PI * 2);
    ctx.strokeStyle = 'rgba(255,255,255,0.05)';
    ctx.lineWidth   = 1;
    ctx.stroke();
    ctx.restore();

    /* ── guide ring ── */
    ctx.save();
    ctx.beginPath();
    ctx.arc(CX, CY, RADIUS, 0, Math.PI * 2);
    ctx.strokeStyle = 'rgba(120,80,200,0.12)';
    ctx.lineWidth   = 6;
    ctx.stroke();
    ctx.restore();

    /* ── arc segments — glassy thick ring with specular edge ── */
    SEGMENTS.forEach(seg => {
      const a0 = angle + seg.start * Math.PI * 2;
      const a1 = a0 + seg.len * Math.PI * 2;

      ctx.save();
      ctx.shadowBlur  = seg.blur * 3;
      ctx.shadowColor = seg.colorA;

      const steps = 64;
      for (let s = 0; s < steps; s++) {
        const frac   = s / steps;
        const aStart = a0 + (a1 - a0) * frac;
        const aEnd   = a0 + (a1 - a0) * (frac + 1 / steps + 0.001);
        const alpha  = seg === SEGMENTS[0]
          ? (1 - frac) * 0.95 + 0.05
          : (1 - frac);

        /* outer glass highlight stroke */
        ctx.beginPath();
        ctx.arc(CX, CY, RADIUS + 2.5, aStart, aEnd);
        ctx.lineWidth   = 1;
        ctx.strokeStyle = seg === SEGMENTS[0]
          ? `rgba(255,255,255,${alpha * 0.35})`
          : `rgba(200,180,255,${alpha * 0.15})`;
        ctx.stroke();

        /* main arc body */
        ctx.beginPath();
        ctx.arc(CX, CY, RADIUS, aStart, aEnd);
        ctx.lineWidth   = seg === SEGMENTS[0] ? 5 : seg.width;
        ctx.strokeStyle = seg === SEGMENTS[0]
          ? `rgba(255,255,255,${alpha * 0.92})`
          : `rgba(160,130,255,${alpha * 0.6})`;
        ctx.stroke();

        /* inner refraction stroke */
        if (seg === SEGMENTS[0]) {
          ctx.beginPath();
          ctx.arc(CX, CY, RADIUS - 3, aStart, aEnd);
          ctx.lineWidth   = 1.5;
          ctx.strokeStyle = `rgba(140,200,255,${alpha * 0.4})`;
          ctx.stroke();
        }
      }
      ctx.restore();

      /* leading dot glow */
      if (seg === SEGMENTS[0]) {
        const lx = CX + Math.cos(a0) * RADIUS;
        const ly = CY + Math.sin(a0) * RADIUS;
        const grd = ctx.createRadialGradient(lx, ly, 0, lx, ly, 11);
        grd.addColorStop(0,   'rgba(220,248,255,1)');
        grd.addColorStop(0.3, 'rgba(140,220,255,0.9)');
        grd.addColorStop(0.7, 'rgba(80,140,255,0.4)');
        grd.addColorStop(1,   'rgba(60,80,200,0)');
        ctx.save();
        ctx.shadowBlur  = 22;
        ctx.shadowColor = '#b0eeff';
        ctx.beginPath();
        ctx.arc(lx, ly, 6.5, 0, Math.PI * 2);
        ctx.fillStyle = grd;
        ctx.fill();
        /* specular highlight on the dot */
        ctx.beginPath();
        ctx.arc(lx - 1.5, ly - 1.5, 2, 0, Math.PI * 2);
        ctx.fillStyle = 'rgba(255,255,255,0.95)';
        ctx.fill();
        ctx.restore();
      }
    });

    /* ── glass sheen sweep (rotates opposite direction) ── */
    ctx.save();
    ctx.globalAlpha = 0.07 + 0.03 * Math.sin(t * 0.025);
    const sheenA0 = -angle * 0.4;
    const sheenA1 = sheenA0 + Math.PI * 0.55;
    ctx.beginPath();
    ctx.arc(CX, CY, RADIUS, sheenA0, sheenA1);
    ctx.lineWidth   = 9;
    ctx.strokeStyle = 'rgba(255,255,255,0.9)';
    ctx.stroke();
    ctx.globalAlpha = 1;
    ctx.restore();

    /* ── center glass orb ── */
    const pulse = 0.18 + 0.06 * Math.sin(t * 0.04);
    const cg = ctx.createRadialGradient(CX - 3, CY - 3, 0, CX, CY, 16);
    cg.addColorStop(0,   `rgba(255,255,255,${pulse * 0.7})`);
    cg.addColorStop(0.3, `rgba(180,150,255,${pulse})`);
    cg.addColorStop(1,   'rgba(0,0,0,0)');
    ctx.beginPath();
    ctx.arc(CX, CY, 16, 0, Math.PI * 2);
    ctx.fillStyle = cg;
    ctx.fill();
    /* tiny specular on orb */
    ctx.beginPath();
    ctx.arc(CX - 3.5, CY - 3.5, 2.5, 0, Math.PI * 2);
    ctx.fillStyle = `rgba(255,255,255,${pulse * 2.2})`;
    ctx.fill();

    requestAnimationFrame(drawFrame);
  }

  /* polyfill: createConicalGradient not standard — we use step approach above */
  drawFrame();
})();
</script>
